fn count_leaves(&self) -> usize

Return the number of leaves.

Example

//               /    \
//            node   leaf
//           /   \
//        leaf   node

use trees::base::QueryableTree;

let mut tree = BinarySearchTree::new();
tree.insert(1);
println!("{}", tree.count_leaves());  // 1
tree.insert(0);
println!("{}", tree.height());  // still 1
